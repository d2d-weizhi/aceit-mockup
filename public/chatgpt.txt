      function openTeamChatModal() {
        document.getElementById('teamChatModal').classList.remove('hidden');
        // Focus the input for accessibility
        document.querySelector('#teamChatModal input[type="text"]').focus();
      }
      // Hide the modal
      document.getElementById('teamChatCloseBtn').onclick = function() {
        document.getElementById('teamChatModal').classList.add('hidden');

      // Optional: ESC closes the modal
      document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape') {
          document.getElementById('teamChatModal').classList.add('hidden');
        }
      });
      // Optional: enable send button when there's text
      const chatInput = document.querySelector('#teamChatModal input[type="text"]');

      if(chatInput && chatSendBtn) {
        chatInput.addEventListener('input', function() {
          chatSendBtn.disabled = this.value.trim() === '';
